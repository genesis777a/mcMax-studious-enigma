# This snippet demonstrates a basic Next.js project structure

import os

# Define the project structure
project_structure = {
    'pages': ['index.js', 'about.js'],
    'public': ['favicon.ico'],
    'styles': ['globals.css']
}

# Function to create directory structure
def create_structure(base_path, structure):
    for dir, files in structure.items():
        dir_path = os.path.join(base_path, dir)
        os.makedirs(dir_path, exist_ok=True)
        for file in files:
            with open(os.path.join(dir_path, file), 'w') as f:
                f.write(f'// {file} content')

# Create the project structure
create_structure('.', project_structure)

print('Next.js project structure created successfully.')
